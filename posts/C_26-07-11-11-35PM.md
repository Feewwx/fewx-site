## 使用函数指针数组重写计算器

```c
int main() {
    int (*arr[4])(int, int) = {Add, Sub, Mul, Div};
    for (int i = 0; i < 4; i++) {
        int ret = arr[i](2, 3);
        printf("%d\n", ret);
    }
    return 0;
}
```
####
```c
#include <stdio.h>

void menu() {
    printf("**********************************\n");
    printf("********  1.add  2.sub  **********\n");
    printf("********  3.mul  4.div  **********\n");
    printf("***********  0.exit  *************\n");
    printf("**********************************\n");
}

int Add(int x, int y) {
    return x + y;
}

int Sub(int x, int y) {
    return x - y;
}

int Mul(int x, int y) {
    return x * y;
}

int Div(int x, int y) {
    return x / y;
}

void calc(int (*pf)(int, int)) {
    int x = 0;
    int y = 0;
    int ret = 0;

    printf("请输入2个操作数:>");
    scanf("%d %d", &x, &y);
    ret = pf(x, y);
    printf("%d\n", ret);
}

int main() {
    int input = 0;
    int x = 0;
    int y = 0;
    int ret = 0;
 
    // 转移表
    int (*pfArr[5])(int, int) = {0, Add, Sub, Mul, Div};

    do {
        menu();
        printf("请选择:>");
        scanf("%d", &input);

        if (input == 0) {
            printf("已退出\n");
        } else if (input >= 1 && input <= 4) {
            printf("请输入2个操作数:>");
            scanf("%d %d", &x, &y);
            ret = pfArr[input](x, y);
            printf("%d\n", ret);
        } else {
            printf("输入错误\n");
        }
    } while (input);
    return 0;
}
```
