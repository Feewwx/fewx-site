## 指针练习

```c
#include <stdio.h>

// =========================================================================
// 1. [类型定义] 定义一个函数指针别名
// =========================================================================
// 原始类型: void (*)(int) -> 代表 "指向一个参数为int, 返回值为void的函数" 的指针.
// 这里的 pf_t 就是这个复杂类型的 "快捷方式" (别名).
typedef void (*pf_t)(int); 


int main() {
    
    // =========================================================================
    // 2. [核心轰炸] 调用绝对地址为 0 处的函数
    // =========================================================================
    // 剥洋葱式拆解:
    // 
    // Step 1:  void (*)()          --> 这是一种类型: 指向 "无参数, 返回void" 的函数的指针.
    // Step 2: (void (*)())0        --> 强转: 把整数 0 强转为上述类型的函数指针.
    //                                  (告诉编译器: 别怀疑, 地址 0 处真的有个函数!)
    // Step 3: *(void (*)())0       --> 解引用: 通过地址取出口子, 准备执行.
    // Step 4: (*(void (*)())0)();  --> 加括号调用: 执行这个函数.
    //
    // * 结果: 在现代系统上运行会报 "段错误" (Segment Fault); 在嵌入式裸机上常用于软复位.
    (*(void (*)())0)();


    // =========================================================================
    // 3. [原生声明] 地狱级难度的 signal 函数局部声明
    // =========================================================================
    // C语言著名的 "右左法则" 实战:
    //
    //                  [2. 它是函数, 接收两个参数: int 和 函数指针]
    //                                  ┌──────┴──────┐
    //            void ( * signal   ( int,  void (*)(int) )   )(int);
    //             └─┬─┘ │  └───┬──┘                             └─┬─┘
    //               │   │      └─> [1. 名字叫 signal]            │
    //               │   └────────────> [3. 返回一个指针]         │
    //               └─────────────────────────────────────────────┴─> [4. 指针指向: 接收int, 返回void的函数]
    //
    // 最终含义: 声明了一个叫 signal 的函数, 它接收一个 int 和一个函数指针作为参数,
    //           并且它执行完后, 还会返回一个相同类型的函数指针.
    void (*signal(int, void (*)(int)))(int);


    // =========================================================================
    // 4. [完美蜕变] 使用别名简化后的相同声明
    // =========================================================================
    // 这行代码的作用和上面那行 [完全一模一样]!
    // 
    // 借助第一行定义的 pf_t, 我们可以轻松读懂:
    // signal 是一个函数 --> 接收一个 int 和一个 pf_t 类型的指针 --> 返回一个 pf_t 类型的指针.
    pf_t signal(int, pf_t);


    return 0;
}
```

## 函数指针的用途

> 写一个计算器

```c
#include <stdio.h>

void menu() {
    printf("**********************************\n");
    printf("********  1.add  2.sub  **********\n");
    printf("********  3.mul  4.div  **********\n");
    printf("***********  0.exit  *************\n");
    printf("**********************************\n");
}

int Add(int x, int y) {
    return x + y;
}

int Sub(int x, int y) {
    return x - y;
}

int Mul(int x, int y) {
    return x * y;
}

int Div(int x, int y) {
    return x / y;
}

// 回调函数
void calc(int (*pf)(int, int)) {
    int x = 0;
    int y = 0;
    int ret = 0;

    printf("请输入2个操作数:>");
    scanf("%d %d", &x, &y);
    ret = pf(x, y);
    printf("%d\n", ret);
}

int main() {
    int input = 0;
    // int x = 0;
    // int y = 0;
    // int ret = 0;

    do {
        menu();
        printf("请选择:>");
        scanf("%d", &input);
        // printf("请输入2个操作数:>");
        // scanf("%d %d", &x, &y);
        // 如果想退出或者输入错误,不能直接退出,必须先输入两个操作数才能退出,很怪
        // 解决方法:把输入两个操作数的步骤塞入case里

        switch (input) {
            case 1:
                // printf("请输入2个操作数:>");
                // scanf("%d %d", &x, &y);
                // ret = Add(x, y);
                // printf("%d\n", ret);
                calc(Add);
                break;
            case 2:
                // printf("请输入2个操作数:>");
                // scanf("%d %d", &x, &y);
                // ret = Sub(x, y);
                // printf("%d\n", ret);
                calc(Sub);
                break;
            case 3:
                // printf("请输入2个操作数:>");
                // scanf("%d %d", &x, &y);
                // ret = Mul(x, y);
                // printf("%d\n", ret);
                calc(Mul);
                break;
            case 4:
                // printf("请输入2个操作数:>");
                // scanf("%d %d", &x, &y);
                // ret = Div(x, y);
                // printf("%d\n", ret);
                calc(Div);
                break;

            // 新问题:代码太过冗余,难维护
            // 解决方法:函数指针

            case 0:
                printf("已退出\n");
                break;
            default:
                printf("输入错误\n");
                break;
        }
    } while (input);
    return 0;
}
```

## 函数指针数组

```c
int main() {
    int (*arr[4])(int, int) = {Add, Sub, Mul, Div};
    for (int i = 0; i < 4; i++) {
        int ret = arr[i](2, 3);
        printf("%d\n", ret);
    }
    return 0;
}
```
