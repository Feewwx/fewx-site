## 一维指针传参

```c
#include <stdio.h>

void print(int *p) {}

int main() {
    int a = 10;
    int *p = &a;
    int arr[10] = {0};
    print(&a);
    print(p);
    print(arr);
```

## 二维指针传参

```c
#include <stdio.h>

test(int **p) {}

int main() {
    int *p1;
    int **p2;
    int *arr[10];
    test(&p1);
    test(p2);
    test(arr);
```

## 函数指针

**数组指针:** - 指向数组的指针就是数组指针

**函数指针:** - 指向函数的指针就是函数指针

```c
#include <stdio.h>

int Add(int x, int y) {
    return x + y;
}

void calc(int (*p)(int, int)) {
    int a = 10;
    int b = 20;
    int ret = p(a, b);
    printf("%d\n", ret);
}

int main() {
    // int arr[5] = {0};
    // &数组名 - 取出数组的地址
    // int (*p)[5] = &arr;

    // &函数名 - 取出函数的地址
    printf("%p\n", &Add);
    // printf("%p\n", Add);
    // 对于函数来说,&函数名和函数名都是函数的地址

    int (*p)(int, int) = &Add;
    // int (*p)(int, int) = Add;
    // int ret = (*p)(2, 3);
    int ret = p(2, 3);
    // int ret = Add(2, 3);
    printf("%d\n", ret);

    calc(Add);
    return 0;
}
```
